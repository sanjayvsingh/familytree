# .htaccess for Family Tree — place in the web root folder
# (same directory as index.php)
# ============================================================

# RewriteEngine On

# ── Block access to sensitive files ──────────────────────────
<FilesMatch "\.(ged|gedcom|json|gitignore)$">
    <IfModule mod_authz_core.c>
        Require all denied
    </IfModule>
    <IfModule !mod_authz_core.c>
        Order deny,allow
        Deny from all
    </IfModule>
</FilesMatch>


# ── Gzip compression ─────────────────────────────────────────
# Compresses JSON API responses by ~75%, JS/CSS by ~70%.
<IfModule mod_deflate.c>
    AddOutputFilterByType DEFLATE application/json
    AddOutputFilterByType DEFLATE application/javascript
    AddOutputFilterByType DEFLATE text/css
    AddOutputFilterByType DEFLATE text/html
    AddOutputFilterByType DEFLATE text/plain
</IfModule>


# ── Security headers ─────────────────────────────────────────
<IfModule mod_headers.c>

    # Tell search engines and crawlers not to index this site
    Header always set X-Robots-Tag "noindex, nofollow, noarchive, nosnippet"

    # Prevent the site from being embedded in iframes (clickjacking)
    Header always set X-Frame-Options "DENY"

    # Stop browsers from guessing content types
    Header always set X-Content-Type-Options "nosniff"

    # Don't leak the URL in Referer headers to third parties
    Header always set Referrer-Policy "no-referrer"

    # Restrict browser features we don't use
    Header always set Permissions-Policy "geolocation=(), camera=(), microphone=()"

    # Content Security Policy:
    # — scripts: self + one inline block (sets window.GEDCOM_FILE)
    # — styles: self + inline (JS sets style attributes on cards)
    # — fetch/XHR: self only (api.php calls)
    # — no external fonts, images, or frames
    Header always set Content-Security-Policy "default-src 'self'; script-src 'self' 'unsafe-inline'; style-src 'self' 'unsafe-inline'; connect-src 'self'; img-src 'self'; frame-ancestors 'none';"

    # Remove the X-Powered-By header that advertises the PHP version
    Header always unset X-Powered-By
    Header unset X-Powered-By

    # No caching for pages that contain family data
    <FilesMatch "\.(php)$">
        Header always set Cache-Control "no-store, no-cache, must-revalidate"
        Header always set Pragma "no-cache"
    </FilesMatch>

    # Cache static assets for 7 days (they change rarely)
    <FilesMatch "\.(js|css)$">
        Header always set Cache-Control "public, max-age=604800"
    </FilesMatch>

</IfModule>


# ── Block common crawler/bot user agents ─────────────────────
<IfModule mod_rewrite.c>
    RewriteEngine On
    RewriteCond %{HTTP_USER_AGENT} (Googlebot|bingbot|Slurp|DuckDuckBot|Baiduspider|YandexBot|Sogou|ia_archiver|AhrefsBot|SemrushBot|MJ12bot|DotBot) [NC]
    RewriteRule .* - [F,L]
</IfModule>
